Class template engine

boost::random::mkl::engine — Class wrapping the intel MKL basic uniform random number generators (BRNG).

Synopsis

// In header: <boost/random/mkl/engine.hpp>

template<int brng = VSL_BRNG_MT19937> 
class engine {
public:
  // types
  typedef int                                                                            result_type;       
  typedef unsigned int                                                                   seed_type;         
  typedef unsigned int                                                                   internal_type;     
  typedef boost::range_generate::unit_stride_range_generator_type_checked< result_type > is_range_generator;

  // construct/copy/destruct
  engine();
  explicit engine(seed_type);
  template<typename SeedSeq> explicit engine(SeedSeq &);
  template<typename It> engine(It, It);
  engine(const engine &);
  engine& operator=(engine);
  ~engine();

  // public member functions
  void seed();
  void seed(seed_type);
  template<typename SeedSeq> void seed(SeedSeq &);
  template<typename It> void seed(It, It);
  void discard(std::size_t);
  template<typename UnitStrideIterator> 
    void operator()(UnitStrideIterator, UnitStrideIterator);
  result_type operator()();
  result_type min() const;
  result_type max() const;
  VSLStreamStatePtr & get_stream();

  // friend functions
  friend bool operator==(engine, engine);
  friend bool operator!=(const engine &, const engine &);
  template<typename CharT, typename Traits> 
    friend std::basic_ostream< CharT, Traits > & 
    operator<<(std::basic_ostream< CharT, Traits > &, const engine &);
  template<typename CharT, typename Traits> 
    friend std::basic_istream< CharT, Traits > & 
    operator>>(std::basic_istream< CharT, Traits > &, engine &);

  // public data members
  static const int id;  // Intel MKL BRNG index. 
  static const unsigned int default_seed;
};

Description

Implements the boost uniform random number generator concept.

Template Parameters

  1. int brng = VSL_BRNG_MT19937

    BRNG index.

engine public types

  1. typedef int result_type;

    Type of the resulting random value.

  2. typedef unsigned int seed_type;

    Type of the seed values.

  3. typedef unsigned int internal_type;

    Internal type used for storing random values. Usually sizeof(internal_type) < sizeof(result_type). In order to find out, how many internal_type values are required for representing one random value, use n_internal() function.

  4. typedef boost::range_generate::unit_stride_range_generator_type_checked< result_type > is_range_generator;

    Required for the engine to model a Range Generator.

engine friend functions

  1. friend bool operator==(engine a, engine b);
    Equality operator.

    Checks whether internal streams are in the same state.

    Note

    Due to unitialized memory in the MKL this method will not rely on comparing MKL stream memory but rather compares the output of both engines for a fixed number of variates and thus is relatively slow.

    Parameters:

    a

    Left hand side instance.

    b

    Right hand side instance.

    Returns:

    Returns true if a and b are in the same state.
