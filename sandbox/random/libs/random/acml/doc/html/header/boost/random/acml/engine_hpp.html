Header <boost/random/acml/engine.hpp>

This file contains a random number engine which wraps functions from the AMD ACML library. 

namespace boost {
  namespace random {
    namespace acml {
      template<long genid_ = 1, bool single = false, long subid_ = 1> 
        class engine;

      template<typename Engine> struct has_block_split;
      template<typename Engine> struct has_leap_frog;
      typedef engine< 1, false, 1 > nag;
      typedef engine< 2, false, 1 > wh;
      typedef engine< 3, false, 1 > mt19937;
      typedef engine< 4, false, 1 > mrg32k3a;
      template<typename Engine, typename Iterator> 
        void make_independent_engines_by_block_split(Iterator, Iterator, 
                                                     const std::size_t &);
      template<typename Engine, typename Iterator> 
        void make_independent_engines_by_leap_frog(Iterator, Iterator);
    }
  }
}
