Header <boost/random/mkl/engine.hpp>

This file contains a random number engine which wraps functions from the Intel MKL library. 

namespace boost {
  namespace random {
    namespace mkl {
      template<int brng = VSL_BRNG_MT19937> class engine;

      template<typename Engine> struct has_block_split;
      template<typename Engine> struct has_leap_frog;
      typedef engine< VSL_BRNG_MCG31 > mcg31;
      typedef engine< VSL_BRNG_R250 > r250;
      typedef engine< VSL_BRNG_MRG32K3A > mrg32k3a;
      typedef engine< VSL_BRNG_MCG59 > mcg59;
      typedef engine< VSL_BRNG_WH > wh;
      typedef engine< VSL_BRNG_MT19937 > mt19937;
      typedef engine< VSL_BRNG_MT2203 > mt2203;
      typedef engine< VSL_BRNG_SFMT19937 > sfmt19937;
      typedef engine< VSL_BRNG_SOBOL > sobol;
      typedef engine< VSL_BRNG_NIEDERR > niederr;
      template<typename Engine, typename Iterator> 
        void make_independent_engines_by_block_split(Iterator, Iterator, 
                                                     std::size_t);
      template<typename Engine, typename Iterator> 
        void make_independent_engines_by_leap_frog(Iterator, Iterator);
    }
  }
}
