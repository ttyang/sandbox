is_tr1_selector

Synopsis 

namespace boost {

    template <typename Selector>
    struct is_tr1_selector
    {
        // typedef ... type;
    };

    template <>
    struct is_tr1_selector<tr1_hash_selector> : ::boost::mpl::true_
    {
    };
}  // namespace boost

Description

Returns boost::mpl::true_ if the specified selector type is tr1_hash_selector, or if the type to be returned by container_gen is a std::tr1::array, a std::tr1::unordered_set, a std::tr1::unordered_multiset, a std::tr1::unordered_map, or a std::tr1::unordered_multimap; boost::mpl::false_ otherwise.

Where defined 

#include <boost/container_gen/is_tr1_selector.hpp>
