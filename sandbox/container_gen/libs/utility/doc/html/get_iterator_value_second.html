        get_iterator_value_second(IterValue& value);

Function Description

Takes in a valid key-value pair in an Associative Container or an associative Boost.PtrContainer; returns a reference to the mapped value.

namespace boost {

    template <typename IterValue>
    struct get_iterator_value_second_result
    {
        typedef typename ::std::tr1::remove_pointer<
                    typename ::boost::unwrap_reference<
                        BOOST_TYPEOF_TPL(
                            ::boost::ref(
                                ::boost::get_reference<
                                    IterValue
                                >().second
                            )
                        )
                    >::type
                >::type&
                type;
    };
}  // namespace boost

Return Type Metafunction Description

Takes in the value type of an Associative Container or an associative Boost.PtrContainer; returns the mapped type.
