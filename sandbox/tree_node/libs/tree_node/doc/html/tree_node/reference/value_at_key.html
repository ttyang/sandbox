result_of::value_at_key<Node,Key>

Synopsis 

namespace boost { namespace tree_node { namespace result_of {

    template <typename Node, typename Key>
    class value_at_key
    {
        typedef typename ::std::tr1::remove_const<Node>::type::super_t
                _base;

     public:
        typedef BOOST_TYPEOF_TPL(
                    get<Key>(
                        ::boost::get_reference<
                            typename ::boost::mpl::if_<
                                ::std::tr1::is_const<Node>
                              , _base const
                              , _base
                            >::type
                        >()
                    )
                )
                type;

    };
}}}  // namespace boost::tree_node::result_of

Description

Let Node be a (possibly const-qualified) Tree Node model. Let Key be a type for which tree_node::result_of::has_key<Node,Key> evaluates to boost::mpl::true_. 

typedef result_of::value_at_key<Node,Key>::type r;

Return type: The actual element type associated with key type Key in Node.

Where defined 

#include <boost/tree_node/intrinsic/value_at_key.hpp>
