EqualityComparable

Description

Describes types whose values can be compared for equality with operator==, which is an equivalence relation.

auto concept EqualityComparable<typename T> : HasEqualTo<T, T> {
  bool operator!=(const T& a, const T& b) { return !(a == b); }

  axiom Consistency(T a, T b) {
    (a == b) == !(a != b);
  }

  axiom Reflexivity(T a) { a == a; }

  axiom Symmetry(T a, T b) {
    if (a == b)
      b == a;
  }

  axiom Transitivity(T a, T b, T c) {
    if (a == b && b == c)
      a == c;
  }
}
