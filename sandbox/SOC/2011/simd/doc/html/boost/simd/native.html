// In header: <boost/simd/sdk/simd/native.hpp>

template<typename Scalar, typename Extension> 
union native {
  // types
  typedef Extension                                extension_type; 
  typedef meta::as_simd< Scalar, Extension >::type native_type;    
  typedef Scalar                                   value_type;     
  typedef value_type &                             reference;      
  typedef value_type const &                       const_reference;
  typedef std::size_t                              size_type;      
  typedef Scalar *                                 iterator;       
  typedef Scalar const *                           const_iterator; 

  // member classes/structs/unions
  template<typename U> 
  struct rebind {
    // types
    typedef native< U, extension_type > type;
  };

  enum @0 { 
            static_size =  sizeof(native_type)/sizeof(value_type)
                       ? sizeof(native_type)/sizeof(value_type) : 1 };

  // construct/copy/destruct
  native& operator=(native const &);
  native& operator=(native_type const &);

  // public member functions
  BOOST_DISPATCH_FORCE_INLINE operator native_type const &() const;
  BOOST_DISPATCH_FORCE_INLINE native_type const & operator()() const;
  BOOST_DISPATCH_FORCE_INLINE iterator begin();
  BOOST_DISPATCH_FORCE_INLINE iterator end();
  BOOST_DISPATCH_FORCE_INLINE const_iterator begin() const;
  BOOST_DISPATCH_FORCE_INLINE const_iterator end() const;
  reference operator[](int);
  const_reference operator[](int) const;
  BOOST_DISPATCH_FORCE_INLINE value_type * data();
  BOOST_DISPATCH_FORCE_INLINE const value_type * data() const;

  // public static functions
  static BOOST_DISPATCH_FORCE_INLINE std::size_t size();

  // public data members
  native_type data_;
  value_type array;
};
