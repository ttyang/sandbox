boost::dispatch::meta::as_real

         >

  // public member functions
   BOOST_MPL_ASSERT_MSG((is_fundamental< typenamemeta::primitive_of< typename meta::strip< T >::type >::type >::value),

                        (T &));
};

Description

Returns the input type rebound with an floating-point type that has the same size as its primitive type.

Models: 

For any type T,

 typedef as_real<T>::type type;

is equivalent to:

  typedef make_real< sizeof(primitive_of<T>::type)
                   , factory_of<T>::type
                   >::type                           type;
