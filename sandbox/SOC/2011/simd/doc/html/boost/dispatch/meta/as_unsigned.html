boost::dispatch::meta::as_unsigned

         >

  // public member functions
   BOOST_MPL_ASSERT_MSG((is_fundamental< typenamemeta::primitive_of< typename meta::strip< T >::type >::type >::value),

                        (T &));
};

Description

Returns the input type rebound the unsigned equivalent type to its primitive type.

Models: 

For any type T,

 typedef as_unsigned<T>::type type;

is equivalent to

 typedef T  type;

if primitive<T>::type is unsigned and to

  typedef apply< meta::factory_of<T>::type
               , boost::make_unsigned< meta::primitive_of<T>::type >::type
               >::type                                                type;

if primitive<T>::type is signed. Note that for this , real types like double and float are considered signed.
