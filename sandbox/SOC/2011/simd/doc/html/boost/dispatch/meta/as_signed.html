boost::dispatch::meta::as_signed

         > 
struct as_signed {

  // public member functions
   BOOST_MPL_ASSERT_MSG((is_fundamental< typenamemeta::primitive_of< typename meta::strip< T >::type >::type >::value),

                        (T &));
};

Description

Returns the input type rebound the signed equivalent type to its primitive type.

Models: 

For any type T,

 typedef as_signed<T>::type type;

is equivalent to

 typedef T  type;

if primitive<T>::type is signed and to

  typedef apply< meta::factory_of<T>::type
               , boost::make_signed< meta::primitive_of<T>::type >::type
               >::type                                              type;

if primitive<T>::type is unsigned. Note than for this , real types like double or float are considered signed.
