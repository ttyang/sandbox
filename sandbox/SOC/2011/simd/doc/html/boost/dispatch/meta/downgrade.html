Struct template downgrade

boost::dispatch::meta::downgrade

Synopsis

// In header: <boost/dispatch/meta/downgrade.hpp>

template<typename T, 
         typename Sign  // Optional sign to use while downgrading Type. If unspecified, Sign is equals to the sign of Type.
         > 
struct downgrade {

  // public member functions
   BOOST_MPL_ASSERT_MSG((is_fundamental< typenamemeta::primitive_of< typename meta::strip< T >::type >::type >::value), 
                        BOOST_DISPATCH_NON_FUNDAMENTAL_PRIMITIVE_USED_IN_META_DOWNGRADE, 
                        (T &));
};

Description

Returns the input type rebound with a type that is twice as small as its primitive, with optional sign change.\

Models: 

 typedef boost::dispatch::meta::downgrade<T, Sign>::type r;

is equivalent to:

 typedef make_floating< max< sizeof(primitive_of<T>::type)/2
                       , sizeof(float)
                       >::value
                  , factory_of<T>::type
                  >::type                                  r;

if primitive_of<T> is of hierarchy boost::dispatch::tag::floating_ and to:

 typedef make_integer< max<sizeof(primitive_of<T>::type)/2, 1>::value
                     , Sign
                     , factory_of<T>::type
                     >::type                                r;

otherwise.
