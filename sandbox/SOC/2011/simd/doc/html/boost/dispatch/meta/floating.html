Struct floating

boost::dispatch::meta::floating

Synopsis

// In header: <boost/dispatch/meta/floating.hpp>


struct floating {
  // member classes/structs/unions
  template<typename This, typename A0> 
  struct result<This(A0)> {
    // types
    typedef float                              base_real;
    typedef meta::make_floating< value >::type type;     

    // public member functions
     BOOST_STATIC_CONSTANT(std::size_t, 
                           value = (sizeof(A0) > sizeof(base_real))?sizeof(A0):sizeof(base_real));
  };
};

Description

For a list of types T0,...,Tn, floating computes the type able to store a value of the smallest floating-point value type compatible with all T0,...,Tn

For types T0,...,Tn,

 typedef result_of<floating(T0,...,Tn)>::type type

is equivalent to:

 typedef boost::dispatch::meta:::make_floating< max<sizeof(float),...,sizeof(Tn)>::value >::type type;
