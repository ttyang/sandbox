Struct template behave_as

boost::dispatch::meta::behave_as

Synopsis

// In header: <boost/dispatch/meta/behave_as.hpp>

template<typename Lambda,   // to apply.
         typename Hierarchizable  // Type to introspect.
         > 
struct behave_as : public boost::mpl::apply1::type< Lambda, primitive_of< strip< Hierarchizable >::type >::type >
{
};

Description

Forward the application of a given Lambda onto the primitive type of a given Hierarchizable.

For any Hierarchizable type T and any Lambda L: 

 typedef boost::dispatch::meta::behave_as<L,T>::type r;

is equivalent to:

 typedef apply1 < L
                , boost::dispatch::meta::primitive_of< boost::dispatch::meta::strip< T >::type >::type
                >::type     r;

For any other types, a BOOST_DISPATCH_BEHAVIOR_OF_NON_FUNDAMENTAL_PRIMITIVE_IS_UNDEFINED static assertion is raised.
